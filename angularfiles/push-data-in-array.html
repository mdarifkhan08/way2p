Push Data In Array

<html ng-app="app">
<head>
<title>Binding Data To html</title>
</head>
<body>
<div ng-controller="nameOfFriends">
<h1>Number of friends={{names.length}}</h1>
<ul>
<li ng-repeat="dost in names">
{{dost.name}}
</li>
</ul>
<form ng-submit="addName()">
<input type="text" placeholder="name" ng-model="bindName">
<button type="submit">Add</button>
</form>
</div>
<script type="text/javascript" src="https://code.angularjs.org/1.5.5/angular.min.js"></script>
<script>
var app=angular.module("app",[]);
app.controller("nameOfFriends",function($scope){
    $scope.names=[{name:'Arif'},{name:'pavan'},{name:'yash alya'}]
    $scope.addName=function(){
        $scope.names.push({name: $scope.bindName});
    }
});
</script>
</body>
</html>
